more_fire_danger = {
	custom_effect_tooltip = more_fire_danger_tt
	hidden_effect = {
		if = {
			limit = {
				has_idea = PBB_small_danger_from_fire
			}
			swap_ideas = {
				remove_idea = PBB_small_danger_from_fire
				add_idea = PBB_danger_from_fire
			}
		}
		else_if = {
			limit = {
				has_idea = PBB_danger_from_fire
			}
			swap_ideas = {
				remove_idea = PBB_danger_from_fire
				add_idea = PBB_big_danger_from_fire
			}
		}
		else_if = {
			limit = {
				has_idea = PBB_big_danger_from_fire
			}
			country_event = pbb.74
		}
		else = {
			add_ideas = PBB_small_danger_from_fire
		}
	}
}

PBB_more_bandit = {
	custom_effect_tooltip = PBB_more_bandit_tt
	hidden_effect = {
		if = {
			limit = {
				has_idea = PBB_very_small_crime
			}
			swap_ideas = {
				remove_idea = PBB_very_small_crime
				add_idea = PBB_small_crime
			}
		}
		else_if = {
			limit = {
				has_idea = PBB_small_crime
			}
			swap_ideas = {
				remove_idea = PBB_small_crime
				add_idea = PBB_crime
			}
		}
		else_if = {
			limit = {
				has_idea = PBB_crime
			}
			swap_ideas = {
				remove_idea = PBB_crime
				add_idea = PBB_big_crime
			}
		}
		else_if = {
			limit = {
				has_idea = PBB_big_crime
			}
			swap_ideas = {
				remove_idea = PBB_big_crime
				add_idea = PBB_very_big_crime
			}
		}
		else_if = {
			limit = {
				has_government = candy
				has_idea = PBB_very_big_crime
			}
			news_event = ooonews.3
			start_civil_war = {
				ideology = neutral
				size = 0.7
					states = all
					states_filter = {
						is_capital = no
					}
				keep_political_leader = no
				keep_political_party_members = yes
			}
		}
		else_if = {
			limit = {
				NOT = { has_government = candy }
				has_idea = PBB_very_big_crime
			}
			news_event = ooonews.3
			start_civil_war = {
				ideology = candy
				size = 0.8
				states = all
				states_filter = {
					is_capital = no
				}
				keep_political_leader = no
				keep_political_party_members = yes
			}
		}
		else = {
			add_stability = -0.05
			add_political_power = -50
		}
	}
}
