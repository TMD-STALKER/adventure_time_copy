upgrade_army_law = {
	if = {
		limit = {
			has_idea = disarmed_nation1
		}
		add_ideas = volunteer_only1
	}
	else_if = {
		limit = {
			has_idea = volunteer_only1
			NOT = { has_idea = only_iam }
		}
		add_ideas = limited_conscription1
	}
	else_if = {
		limit = {
			has_idea = limited_conscription1
			NOT = { has_idea = only_iam }
		}
		add_ideas = extensive_conscription1
	}
	else_if = {
		limit = {
			has_idea = extensive_conscription1
			NOT = { has_idea = only_iam }
		}
		add_ideas = service_by_requirement1
	}
	#else_if = {
	#	limit = {
	#		has_idea = service_by_requirement1
	#	}
	#	add_ideas = all_adults_serve1
	#}
	else_if = {
		limit = {
			has_idea = GENERIC_snow_army
		}
		add_manpower = 15255
	}
	else = {
		add_manpower = 150
		add_political_power = 150
	}
}
upgrade_economy_law = {
	if = {
		limit = {
			OR = {
				has_idea = PBB_verymuch_social_spend
				has_idea = icecream_policy
			}
		}
		add_war_support = 0.1
		add_political_power = 50
		army_experience = 10
	}
	else_if = {
		limit = {
			has_idea = civilian_economy
		}
		add_ideas = low_economic_mobilisation
	}
	else_if = {
		limit = {
			has_idea = low_economic_mobilisation
		}
		add_ideas = partial_economic_mobilisation
	}
	else_if = {
		limit = {
			has_idea = partial_economic_mobilisation
		}
		add_ideas = war_economy
	}
	else_if = {
		limit = {
			has_idea = war_economy
		}
		add_ideas = tot_economic_mobilisation
	}
	else = {
		add_political_power = 150
	}
}
